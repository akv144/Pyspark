        lr=0.05
        n=len(x)

        for i in range(1000):
            yp=m_curr*x+b_curr
            plt.plot(x,yp,color='green')
            
            md=-(2/n)*sum(x*(y-yp))
            bd=-(2/n)*sum((y-yp))
            m_curr=m_curr-lr*md
